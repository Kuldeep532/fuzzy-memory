name: Build NexusPlus APK
on:
  workflow_dispatch: # Ab ye workflow sirf aapke manually start karne par chalega
jobs:
  build:
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: NexusPlus
    steps:
      - name: Checkout Repository
        uses: actions/checkout@v4
      - name: Setup Java 17
        uses: actions/setup-java@v4
        with:
          distribution: temurin
          java-version: 17
      - name: Setup Android SDK
        uses: android-actions/setup-android@v3
      - name: Setup Gradle Cache
        uses: gradle/actions/setup-gradle@v4
      - name: Validate Required Secrets
        run: |
          test -n "${{ secrets.GOOGLE_SERVICES_JSON }}" || (echo "Missing GOOGLE_SERVICES_JSON" && exit 1)
          test -n "${{ secrets.KEYSTORE_BASE64 }}" || (echo "Missing KEYSTORE_BASE64" && exit 1)
          test -n "${{ secrets.KEYSTORE_PASSWORD }}" || (echo "Missing KEYSTORE_PASSWORD" && exit 1)
          test -n "${{ secrets.KEY_ALIAS }}" || (echo "Missing KEY_ALIAS" && exit 1)
          test -n "${{ secrets.KEY_PASSWORD }}" || (echo "Missing KEY_PASSWORD" && exit 1)
      - name: Restore google-services.json
        run: |
          mkdir -p app
          printf '%s' '${{ secrets.GOOGLE_SERVICES_JSON }}' > app/google-services.json
      - name: Restore Keystore
        run: |
          printf '%s' "${{ secrets.KEYSTORE_BASE64 }}" | base64 --decode > release.keystore
      - name: Create key.properties
        env:
          STORE_PASSWORD: ${{ secrets.KEYSTORE_PASSWORD }} # GitHub safely env var में डालता है
        run: |
          cat > key.properties <<EOF_PROPERTIES
          storeFile=release.keystore
          EOF_PROPERTIES
          printf '%s\n' "storePassword=$STORE_PASSWORD" >> key.properties # shell variable expand होता है, content नहीं
          cat >> key.properties <<EOF_PROPERTIES
          keyAlias=${{ secrets.KEY_ALIAS }}
          keyPassword=${{ secrets.KEY_PASSWORD }}
          EOF_PROPERTIES
      - name: Diagnostics
        run: |
          java -version
          ./gradlew --version
          test -f key.properties
          test -f release.keystore
          test -f app/google-services.json
      - name: Grant Gradle Permission
        run: chmod +x gradlew
      - name: Build Debug APK # --warning-mode none aur --quiet lagaya hai taaki faltu warnings na dikhein
        run: ./gradlew assembleDebug --quiet --warning-mode none --no-daemon
      - name: Build Release APK # --info hata kar --quiet aur --warning-mode none kiya hai taaki sirf main error dikhe
        run: ./gradlew clean assembleRelease --quiet --warning-mode none --no-daemon
      - name: Upload Debug APK
        uses: actions/upload-artifact@v4
        with:
          name: NexusPlus-Debug
          path: NexusPlus/app/build/outputs/apk/debug/*.apk
      - name: Upload Release APK
        uses: actions/upload-artifact@v4
        with:
          name: NexusPlus-Production
          path: NexusPlus/app/build/outputs/apk/release/*.apk
